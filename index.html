    Question 4: Write a code to calculate area of a rectangle using inner function. In this case outer function should accept parameter length and inner function 
    should accept parameter breadth.
    
    (function outer(length) {
        (function inner(breadth) {
            let area = length * breadth
            console.log( `Area of Rectangle of length ${length} and breadth ${breadth} is = ${area}`);
        })(30);
    })(12);
